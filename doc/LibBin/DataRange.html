def initialize(*args)
  if args.length == 2
    @range = Range::new(args[0], args[1])
  else
    if args[0].kind_of?(Hash)
      @range = args[0].values.compact.collect(&:range).reduce(:+)
    else
      @range = args[0].compact.collect(&:range).reduce(:+)
    end
  end
end
